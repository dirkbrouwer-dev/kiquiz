  <script>
    const correctAnswers = [1956, 30, 97, 93, 10]; // Richtige Antworten für die Fragen

    function checkAnswers() {
      const userAnswers = [];
      const button = document.getElementById('submitButton');

      // Add animation to the button
      button.classList.add('animate-pulse');

      // Simulate a delay for the animation
      setTimeout(() => {
        for (let i = 1; i <= correctAnswers.length; i++) {
          const answer = parseFloat(document.getElementById(`answer${i}`).value);
          userAnswers.push(answer);
        }

        userAnswers.forEach((answer, index) => {
          const correct = correctAnswers[index];
          const diff = Math.abs(correct - answer);
          const resultBox = document.getElementById(`result${index + 1}`);

          resultBox.innerHTML = `<p class='mt-2 text-sm'><span class='font-semibold'>Deine Antwort:</span> <span class='text-blue-500'>${answer}</span><br>
            <span class='font-semibold'>Richtige Antwort:</span> <span class='text-green-500'>${correct}</span><br>
            <span class='font-semibold'>Abweichung:</span> <span class='text-red-500'>${diff}</span></p>`;
          resultBox.classList.remove("hidden");
        });

        // Remove animation after results are displayed
        button.classList.remove('animate-pulse');
      }, 500);
    }
  </script>
